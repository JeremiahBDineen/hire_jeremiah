map_color= function(x) {
  anchor_values= c(0, 20000, 100000, 700000)
  anchor_colors= c("#FFFF00", "#FFA500", "#FF0000", "#800080")
  anchor_rgb= col2rgb(anchor_colors) / 255
  color_result= character(length(x))
  
  for (i in seq_along(x)) {
    val= x[i]
    if (is.na(val)) {
      color_result[i]= '#BEBEBE'
      next
    }
    
    val_clamped= max(min(val, max(anchor_values)), min(anchor_values))
    zone= findInterval(val_clamped, anchor_values, rightmost.closed = TRUE)
    x0= anchor_values[zone]
    x1= anchor_values[zone + 1]
    t= (val_clamped - x0) / (x1 - x0)
    rgb_start= anchor_rgb[, zone]
    rgb_end= anchor_rgb[, zone + 1]
    rgb_interp= (1 - t) * rgb_start + t * rgb_end
    rgb_interp= pmin(pmax(rgb_interp, 0), 1)
    color_result[i]= rgb(rgb_interp[1], rgb_interp[2], rgb_interp[3])
  }
  return(color_result)
}

county_data=
  readRDS('data/us_county.rds') %>%
  filter(cohort=='household', data_year == '2023') %>%
  arrange(desc(household_count)) %>%
  mutate(
    cum_population= cumsum(household_count)/sum(household_count, na.rm=T),
    cum_county= 1:n()/n(),
    household_size= factor(case_when(
      household_count > 600e3 ~ '> 600K',
      household_count > 100e3 ~ '> 100K',
      household_count >  20e3 ~ '>  20K',
      household_count <= 20e3 ~ '<= 20K',
      .default= NA),
      ordered=T,
      levels= c('> 600K', '> 100K', '>  20K', '<= 20K')),
    household_gradient= map_color(household_count))

write.csv(
  county_data %>%
    mutate(
      household_count= formatC(household_count, format= 'd', big.mark= ','),
      cum_county= paste0(round(100*cum_county, 2), '%'),
      cum_population= paste0(round(100*cum_population, 2), '%')
      ) %>%
    select(county, state, household_count, cum_county, cum_population) %>%
    filter(!is.na(county)),
  'output/tables/population_table.csv',
  row.names= F)
